this is the first file to practice git.

git branch -- shows branches
git branch branch2 -- created a new branch that is branch2
git switch branch2 -- head points to branch2
git switch -c dark-mode  -- creates and switch to that command only
git checkout orange-mode -- used to create a branch

git merge branch2 -- merges the content
git merge --abort -- to abort conflicting merge

git branch -m branch2 dark-mode -- used to rename branch

git branch -d branch2 -- used to delete a branch

git diff -- does comparison btw two entites
git diff --staged --- describes the comparison btw staging area
git diff <branch1>..<branch2>  --- does comparison b'w two branches

git stash -- temporary location to store uncommited work, when branch switching required
git stash apply -- used to recover from stash, last element put in stash will be recovered first.
git stash list --- used to check list of stashes
git stash apply stash@{0}  --- used to import perticular no stash
git stash pop --- stash apply as well as pop from list
git stash drop --- drop stash from list
